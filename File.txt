  цел n, i, x, sum  
  логический флаг  
  строка текст  

  вывод "Введите число (сколько раз выполнять цикл): "  
  ввод n

  текст := "Работаем со всеми элементами Кумир"  

  вывод "Введите число для вычислений: "  
  ввод x  
  sum := 0  

  нц для i от 1 до n  
    sum := sum + i * x  
  кц  

  вывод "Сумма выражений (1*x + 2*x + ... + n*x) = ", sum  

  если sum > 100 то  
    вывод "Сумма больше 100"  
    флаг := истина  
  иначе  
    вывод "Сумма меньше или равна 100"  
  все  

  вывод "Введите команду (1 - приветствие, 2 - работа с роботом): "  
  ввод i  

  выбор i  
    1:  
      вывод "Привет! Вот текст: ", текст

      нц раз 3  
        выбор  
          при справа свободно:  
            вправо  
          при снизу свободно:  
            вниз  
          иначе  
            вверх; влево  
        все  
        закрасить  
      кц  
  все  

  вывод "Введите число для цикла пока: "  
  ввод x  

  нц пока x > 0  
    вывод "x = ", x  
    x := x - 1  
  кц
